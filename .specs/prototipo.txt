// --- ⚙️ MOTOR INDEXED_DB (Para datos pesados) ---
const DB_NAME = 'AquaInfraDB';
const DB_VERSION = 1;
const STORE_NAME = 'mediciones_base';

const getDB = (): Promise<IDBDatabase> => {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onerror = () => reject(request.error);
    request.onsuccess = () => resolve(request.result);
    request.onupgradeneeded = (e: any) => {
      const db = e.target.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        db.createObjectStore(STORE_NAME);
      }
    };
  });
};

const saveBaseRecords = async (id: string, records: any[]) => {
  try {
    const db = await getDB();
    const tx = db.transaction(STORE_NAME, 'readwrite');
    tx.objectStore(STORE_NAME).put(records, id);
    return new Promise((resolve, reject) => { tx.oncomplete = resolve; tx.onerror = reject; });
  } catch (e) { console.error("Error IndexedDB save", e); }
};

const getBaseRecords = async (id: string): Promise<any[]> => {
  try {
    const db = await getDB();
    const tx = db.transaction(STORE_NAME, 'readonly');
    const req = tx.objectStore(STORE_NAME).get(id);
    return new Promise((resolve, reject) => {
      req.onsuccess = () => resolve(req.result || []);
      req.onerror = () => reject(req.error);
    });
  } catch (e) { console.error("Error IndexedDB get", e); return []; }
};

const deleteBaseRecords = async (id: string) => {
  try {
    const db = await getDB();
    const tx = db.transaction(STORE_NAME, 'readwrite');
    tx.objectStore(STORE_NAME).delete(id);
  } catch (e) { console.error("Error IndexedDB delete", e); }
};

export function agruparPorKm(records: any[], isCrudo: boolean, condPuntual: number, condMedio: number) {
  if (records.length === 0) return { stats: null, km: [] };
  
  let maxPuntual = -Infinity;
  let fallosGlobales = 0; 
  let sumMediaGlobal = 0; 
  let kmFallidos = 0;
  const combinedKm: IRI_ResumenKm[] = [];

  let currentKmStart = records[0].from_m;
  let currentKmEnd = currentKmStart + 1000;
  let currentKmRecords: any[] = [];

  const processKm = () => {
    if (currentKmRecords.length > 0) {
      let fallosKm = 0; let sumKm = 0;
      currentKmRecords.forEach(r => {
         let val = 0;
         let isFallo = false;
         if (isCrudo) {
            val = Math.max(r.iri_izq, r.iri_der);
            isFallo = r.iri_izq > condPuntual || r.iri_der > condPuntual;
            sumKm += r.promedio;
         } else {
            val = r.valor;
            isFallo = r.valor > condPuntual;
            sumKm += r.valor;
         }
         if (val > maxPuntual) maxPuntual = val;
         if (isFallo) fallosKm++;
      });

      const valor_medio = sumKm / currentKmRecords.length;
      const cumple_medio = valor_medio <= condMedio;
      const cumple_km = cumple_medio && fallosKm === 0;

      fallosGlobales += fallosKm;
      sumMediaGlobal += valor_medio;
      if(!cumple_km) kmFallidos++;

      const kf = Math.floor(currentKmStart / 1000);
      const rf = String(Math.round(currentKmStart % 1000)).padStart(3, '0');
      const kt = Math.floor(currentKmEnd / 1000);
      const rt = String(Math.round(currentKmEnd % 1000)).padStart(3, '0');

      combinedKm.push({
         km_label: `K${kf}+${rf} - K${kt}+${rt}`,
         from_m: currentKmStart,
         to_m: currentKmEnd,
         valor_medio,
         n_puntuales: currentKmRecords.length,
         n_incumplimientos: fallosKm,
         cumple_medio,
         cumple_km
      });
    }
  };

  for (let i = 0; i < records.length; i++) {
    const r = records[i];
    while (r.from_m >= currentKmEnd) {
      processKm();
      currentKmStart = currentKmEnd;
      currentKmEnd += 1000;
      currentKmRecords = [];
    }
    currentKmRecords.push(r);
  }
  processKm(); 

  const stats: IRI_Estadisticas = {
     total_registros_base: records.length, 
     total_puntuales: records.length,
     valor_medio_global: combinedKm.length ? sumMediaGlobal / combinedKm.length : 0,
     valor_max_puntual: maxPuntual === -Infinity ? 0 : maxPuntual,
     valor_min_puntual: 0, 
     n_incumplimientos_puntual: fallosGlobales,
     n_km_incumplen: kmFallidos,
     porcentaje_cumplimiento: records.length ? ((records.length - fallosGlobales)/records.length)*100 : 0,
     cumple_global: kmFallidos === 0
  };

  return { stats, km: combinedKm };
}

export function parsearArchivoIRI(
  csvText: string,
  condicionPuntual: number,
  condicionMedio: number
): DatosIRI {
  const lines = csvText.split(/\r?\n/).filter(l => l.trim().length > 0);
  if (lines.length < 2) throw new Error("Archivo vacío o sin datos.");

  const parseDistancia = (str: string): number => {
    if (!str) return NaN;
    const clean = str.trim().replace(/"/g, '');
    const prMatch = clean.toUpperCase().match(/PR(\d+)\+(\d+)/);
    if (prMatch) return parseInt(prMatch[1]) * 1000 + parseInt(prMatch[2]);
    const kMatch = clean.toUpperCase().match(/K(\d+)\+(\d+)/);
    if (kMatch) return parseInt(kMatch[1]) * 1000 + parseInt(kMatch[2]);
    return parseFloat(clean.replace(/,/g, '.'));
  };

  const parseNum = (str: string): number => {
    if (!str) return NaN;
    return parseFloat(str.trim().replace(/"/g, '').replace(/,/g, '.'));
  };

  const splitCSV = (line: string): string[] => {
    const result: string[] = [];
    let inQuote = false;
    let current = '';
    for (const c of line) {
      if (c === '"') { inQuote = !inQuote; }
      else if ((c === ',' || c === ';' || c === '\t') && !inQuote) {
        result.push(current.trim()); current = '';
      } else { current += c; }
    }
    result.push(current.trim());
    return result;
  };

  let dataStartIndex = 0;
  let intervalo_base_m = 20; 

  for (let i = 0; i < Math.min(10, lines.length); i++) {
    const cols = splitCSV(lines[i]);
    if (cols.length >= 5) {
      const testFrom = parseDistancia(cols[0]);
      const testIri = parseNum(cols[3]);
      if (!isNaN(testFrom) && !isNaN(testIri)) {
        dataStartIndex = i;
        const parseDist = parseNum(cols[2]);
        if (!isNaN(parseDist) && parseDist > 0) intervalo_base_m = parseDist;
        break;
      }
    }
  }

  const registros_base: IRI_RegistroBase[] = [];
  for (let i = dataStartIndex; i < lines.length; i++) {
    const cols = splitCSV(lines[i]);
    if (cols.length < 5) continue;

    const from_m  = parseDistancia(cols[0]);
    const to_m    = parseDistancia(cols[1]);
    const iri_izq = parseNum(cols[3]);
    const iri_der = parseNum(cols[4]);

    if (isNaN(from_m) || isNaN(to_m) || isNaN(iri_izq) || isNaN(iri_der)) continue;

    const promedio = (iri_izq + iri_der) / 2;
    registros_base.push({ from_m, to_m, iri_izq, iri_der, promedio });
  }

  if (registros_base.length === 0) throw new Error("No se encontraron datos válidos. Verifica las columnas.");

  const pointsPer100m = Math.max(1, Math.round(100 / intervalo_base_m));
  const registros_puntual: IRI_RegistroPuntual[] = [];
  
  for (let i = 0; i < registros_base.length; i += pointsPer100m) {
    const chunk = registros_base.slice(i, i + pointsPer100m);
    if (chunk.length === 0) continue;
    const valor = chunk.reduce((acc, r) => acc + r.promedio, 0) / chunk.length;
    registros_puntual.push({
      from_m: chunk[0].from_m,
      to_m:   chunk[chunk.length - 1].to_m,
      valor,
      cumple_puntual: valor <= condicionPuntual
    });
  }

  const { stats, km } = agruparPorKm(registros_puntual, false, condicionPuntual, condicionMedio);
  if (stats) stats.total_registros_base = registros_base.length;

  return {
    tipo: 'IRI', version: '1.0', unidad: 'm/km', intervalo_base_m, intervalo_puntual_m: 100,
    registros_base, registros_puntual, 
    resumen_km: km || [], 
    estadisticas: stats || {
       total_registros_base: 0, total_puntuales: 0, valor_medio_global: 0,
       valor_max_puntual: 0, valor_min_puntual: 0, n_incumplimientos_puntual: 0,
       n_km_incumplen: 0, porcentaje_cumplimiento: 0, cumple_global: true
    }
  };
}

const appReducer = (state: AppState, action: Action): AppState => {
  switch (action.type) {
    case 'SET_MODULE': return { ...state, activeModule: action.payload };
    case 'SET_ACTIVE_PROJECT': return { ...state, activeProjectId: action.payload };
    case 'ADD_PROYECTO': return { ...state, proyectos: [...state.proyectos, action.payload] };
    case 'UPDATE_PROYECTO': return { ...state, proyectos: state.proyectos.map(p => p.id === action.payload.id ? action.payload : p) };
    case 'DELETE_PROYECTO': return { ...state, proyectos: state.proyectos.filter(p => p.id !== action.payload), activeProjectId: state.activeProjectId === action.payload ? null : state.activeProjectId };
    case 'ADD_TAG': return { ...state, configuracionTags: [...state.configuracionTags, action.payload] };
    case 'UPDATE_TAG': return { ...state, configuracionTags: state.configuracionTags.map(t => t.id === action.payload.id ? action.payload : t) };
    case 'DELETE_TAG': return { ...state, configuracionTags: state.configuracionTags.filter(t => t.id !== action.payload) };
    case 'ADD_UF': return { ...state, unidadesFuncionales: [...state.unidadesFuncionales, action.payload] };
    case 'UPDATE_UF': return { ...state, unidadesFuncionales: state.unidadesFuncionales.map(u => u.id === action.payload.id ? action.payload : u) };
    case 'DELETE_UF': return { ...state, unidadesFuncionales: state.unidadesFuncionales.filter(u => u.id !== action.payload) };
    case 'SAVE_RUTAS_BATCH': return { ...state, rutas: [...state.rutas.filter(r => r.unidadFuncionalId !== action.payload.ufId), ...action.payload.rutas] };
    case 'ADD_INDICADOR': return { ...state, indicadores: [...state.indicadores, { ...action.payload, id: `ind_${generateId()}`, frecuencia_dias: FRECUENCIAS[action.payload.frecuencia].dias, createdAt: new Date().toISOString(), updatedAt: new Date().toISOString() } as Indicador] };
    case 'UPDATE_INDICADOR': return { ...state, indicadores: state.indicadores.map(i => i.id === action.payload.id ? {...action.payload, updatedAt: new Date().toISOString()} : i) };
    case 'DELETE_INDICADOR': return { ...state, indicadores: state.indicadores.filter(i => i.id !== action.payload) };
    case 'ADD_MEDICION': return { ...state, mediciones: [action.payload, ...state.mediciones] };
    case 'DELETE_MEDICION': { deleteBaseRecords(action.payload); return { ...state, mediciones: state.mediciones.filter(m => m.id !== action.payload) }; }
    default: return state;
  }
};

const InlineEdit = ({ value, onSave }: { value: string, onSave: (val: string) => void }) => {
  const [isEditing, setIsEditing] = useState(false);
  const [tempValue, setTempValue] = useState(value);
  const handleSave = () => { if (tempValue.trim() !== '') { onSave(tempValue); } else { setTempValue(value); } setIsEditing(false); };
  if (isEditing) {
    return (
      <div className="flex items-center gap-2">
        <Input autoFocus value={tempValue} onChange={(e: any) => setTempValue(e.target.value)} className="h-8 text-sm w-full" onKeyDown={(e: any) => e.key === 'Enter' && handleSave()} />
        <Button variant="primary" size="icon" className="h-8 w-8 shrink-0" onClick={handleSave}><Check className="w-4 h-4" /></Button>
      </div>
    );
  }
  return (
    <div className="flex items-center justify-between group cursor-pointer" onClick={() => setIsEditing(true)}>
      <span className="truncate">{value}</span>
      <Edit2 className="w-4 h-4 text-slate-400 opacity-0 group-hover:opacity-100 transition-opacity" />
    </div>
  );
};

const TagMultiSelect = ({ value, onChange, disabled = false, options }: { value: string[], onChange: (val: string[]) => void, disabled?: boolean, options: ConfiguracionTag[] }) => {
  const handleToggle = (tagObj: ConfiguracionTag) => {
    if (disabled) return;
    let newValue = [...value];
    if (newValue.includes(tagObj.tag)) { newValue = newValue.filter(t => t !== tagObj.tag); } 
    else {
      if (tagObj.tipo === 'bifurcado') { const unicos = options.filter(o => o.tipo === 'unico').map(o => o.tag); newValue = newValue.filter(t => !unicos.includes(t)); } 
      else if (tagObj.tipo === 'unico') { const bifurcados = options.filter(o => o.tipo === 'bifurcado').map(o => o.tag); newValue = newValue.filter(t => !bifurcados.includes(t)); }
      newValue.push(tagObj.tag);
    }
    onChange(newValue);
  };
  return (
    <div className="flex flex-wrap gap-2">
      {options.map(opt => {
        const isSelected = value.includes(opt.tag);
        const hasUnico = value.some(t => options.find(o => o.tag === t)?.tipo === 'unico');
        const hasBifurcado = value.some(t => options.find(o => o.tag === t)?.tipo === 'bifurcado');
        let isDisabledOption = disabled;
        if (!isSelected && !disabled) {
           if (opt.tipo === 'bifurcado' && hasUnico) isDisabledOption = true;
           if (opt.tipo === 'unico' && hasBifurcado) isDisabledOption = true;
        }
        return (
          <button key={opt.id} type="button" onClick={() => handleToggle(opt)} disabled={isDisabledOption} title={opt.descripcion}
            className={`px-2 py-1 text-xs font-bold rounded border transition-all ${isSelected ? 'bg-[#FFF3CD] border-[#FFE69C] text-[#664D03] shadow-sm' : 'bg-slate-50 border-slate-200 text-slate-500 hover:bg-slate-100'} ${isDisabledOption ? 'opacity-40 cursor-not-allowed' : 'cursor-pointer'}`}>
            {opt.tag}
          </button>
        );
      })}
    </div>
  );
};

const IndicadorForm = ({ isOpen, onClose, indicadorEdit = null }: { isOpen: boolean, onClose: () => void, indicadorEdit?: Indicador | null }) => {
  const { state, dispatch } = useAppStore();
  const [error, setError] = useState<string | null>(null);
  const [form, setForm] = useState<Partial<Indicador>>({ nombre: '', identificador: '', concepto_medicion: '', normatividad: '', metodo_medida: '', frecuencia: 'MENSUAL', unidad_medida: '', tiempo_correccion_valor: null, tiempo_correccion_unidad: 'N/A', condiciones: [] });
  const [isAddingCond, setIsAddingCond] = useState(false);
  const [newCond, setNewCond] = useState<Partial<CondicionIndicador>>({ nombre: '', tipo: 'PUNTUAL', operador: '<=', valor: 0, unidad: '' });

  useEffect(() => {
    if (isOpen) {
      if (indicadorEdit) setForm({ ...indicadorEdit });
      else setForm({ nombre: '', identificador: '', concepto_medicion: '', normatividad: '', metodo_medida: '', frecuencia: 'MENSUAL', unidad_medida: '', tiempo_correccion_valor: null, tiempo_correccion_unidad: 'N/A', condiciones: [] });
      setError(null); setIsAddingCond(false);
    }
  }, [isOpen, indicadorEdit]);

  const handleSave = () => {
    if (!form.nombre?.trim() || !form.identificador?.trim()) return setError("Nombre e identificador son obligatorios.");
    if (!form.condiciones || form.condiciones.length === 0) return setError("Agregue al menos 1 condición.");
    const isDuplicate = state.indicadores.some(i => i.identificador.toLowerCase() === form.identificador?.toLowerCase() && i.id !== indicadorEdit?.id);
    if (isDuplicate) return setError(`Identificador "${form.identificador}" ya existe.`);
    if (indicadorEdit) dispatch({ type: 'UPDATE_INDICADOR', payload: form as Indicador });
    else dispatch({ type: 'ADD_INDICADOR', payload: form as any });
    onClose();
  };

  const handleAddCondicion = () => {
    if (!newCond.nombre?.trim()) return;
    const cond: CondicionIndicador = { id: generateId(), nombre: newCond.nombre, tipo: newCond.tipo as TipoCondicion, operador: newCond.operador as OperadorCondicion, valor: Number(newCond.valor), valor_max: newCond.operador === 'between' ? Number(newCond.valor_max) : undefined, unidad: newCond.unidad || form.unidad_medida || '' };
    setForm(prev => ({ ...prev, condiciones: [...(prev.condiciones || []), cond] }));
    setIsAddingCond(false); setNewCond({ nombre: '', tipo: 'PUNTUAL', operador: '<=', valor: 0, unidad: form.unidad_medida || '' });
  };

  return (
    <SlidePanel isOpen={isOpen} onClose={onClose} title={indicadorEdit ? "Editar Indicador" : "Nuevo Indicador"} customWidth="w-full sm:w-[680px]">
      <div className="p-6 space-y-8 pb-24">
        {error && <div className="p-3 bg-red-50 text-red-600 rounded-md border flex items-center gap-2"><AlertTriangle className="w-4 h-4" /> {error}</div>}

const NuevaMedicionWizard = () => {
  const { state, dispatch, changeView } = useAppStore();
  const [step, setStep] = useState(1);
  const [form, setForm] = useState({ proyectoId: state.activeProjectId || '', ufId: '', rutaId: '', indicadorId: '', fecha: new Date().toISOString().split('T')[0] });
  const [carrilesSeleccionados, setCarrilesSeleccionados] = useState<string[]>([]);
  const [parsedData, setParsedData] = useState<DatosIRI | null>(null);
  const [isProcessing, setIsProcessing] = useState(false);
  const [processError, setProcessError] = useState<string | null>(null);

  const ufs = state.unidadesFuncionales.filter(u => u.proyectoId === form.proyectoId);
  const rutas = state.rutas.filter(r => r.unidadFuncionalId === form.ufId);
  const rutaActiva = state.rutas.find(r => r.id === form.rutaId);
  const indicadorActivo = state.indicadores.find(i => i.id === form.indicadorId);

  const toggleCarril = (tag: string) => carrilesSeleccionados.includes(tag) ? setCarrilesSeleccionados(prev => prev.filter(t => t !== tag)) : setCarrilesSeleccionados(prev => [...prev, tag]);

  const handleFileUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;
    setIsProcessing(true); setProcessError(null); setParsedData(null);

    const reader = new FileReader();
    reader.onload = (event) => {
      try {
        const text = event.target?.result as string;
        if (!indicadorActivo) throw new Error("No hay indicador");
        
        const condPuntual = indicadorActivo.condiciones.find(c => c.tipo === 'PUNTUAL')?.valor || 
                            indicadorActivo.condiciones.find(c => c.tipo === 'GENERAL')?.valor || 3.5;
        const condMedio = indicadorActivo.condiciones.find(c => c.tipo === 'MEDIO')?.valor || 
                          indicadorActivo.condiciones.find(c => c.tipo === 'GENERAL')?.valor || 3.0;

        setParsedData(parsearArchivoIRI(text, condPuntual, condMedio));
      } catch (err: any) { setProcessError(err.message || "Error al parsear."); } 
      finally { setIsProcessing(false); }
    };
    reader.readAsText(file);
  };

  const handleSaveAll = async () => {
    if (!parsedData || !indicadorActivo) return;
    const { registros_base, ...lightData } = parsedData;
    for (const carril of carrilesSeleccionados) {
      const medicionId = `med_${generateId()}`;
      await saveBaseRecords(medicionId, registros_base);
      dispatch({ type: 'ADD_MEDICION', payload: { id: medicionId, proyectoId: form.proyectoId, unidadFuncionalId: form.ufId, rutaId: form.rutaId, carrilesSeleccionados: [carril], indicadorId: form.indicadorId, fecha: form.fecha, datos: lightData, tiene_datos_base: true, createdAt: new Date().toISOString(), updatedAt: new Date().toISOString() } });
    }
    changeView('mediciones_list');
  };

  return (
    <div className="max-w-4xl mx-auto space-y-6 pb-20">
      <div className="flex items-center gap-4 mb-8"><Button variant="ghost" size="icon" onClick={() => changeView('mediciones_list')}><ChevronLeft className="w-5 h-5" /></Button><div><h1 className="text-2xl font-bold">Nueva Medición</h1></div></div>
      <div className="flex items-center justify-between relative mb-12 px-4">
        <div className="absolute left-8 right-8 top-1/2 h-1 bg-slate-200 -z-10 -translate-y-1/2 rounded-full"></div>
        <div className="absolute left-8 top-1/2 h-1 bg-[#00C2D4] -z-10 -translate-y-1/2 transition-all duration-500" style={{ width: `${((step - 1) / 3) * 100}%` }}></div>
        {[1, 2, 3, 4].map(s => <div key={s} className={`w-10 h-10 rounded-full flex items-center justify-center font-bold border-4 ${step >= s ? 'bg-[#00C2D4] border-white text-[#0A1628]' : 'bg-slate-100 border-white text-slate-400'}`}>{step > s ? <Check className="w-5 h-5" /> : s}</div>)}
      </div>

      {step === 1 && (
        <Card className="p-6 animate-in fade-in">
          <h2 className="text-lg font-bold mb-6 border-b pb-2">1. Contexto</h2>
          <div className="grid grid-cols-2 gap-8"><div className="space-y-4">
              <div><label className="text-xs font-semibold block mb-1">Proyecto</label><Select value={form.proyectoId} onChange={(e:any) => setForm({...form, proyectoId: e.target.value, ufId: '', rutaId: ''})}><option value="">Seleccione...</option>{state.proyectos.map(p => <option key={p.id} value={p.id}>{p.nombre}</option>)}</Select></div>
              <div><label className="text-xs font-semibold block mb-1">UF</label><Select value={form.ufId} onChange={(e:any) => setForm({...form, ufId: e.target.value, rutaId: ''})} disabled={!form.proyectoId}><option value="">Seleccione...</option>{ufs.map(u => <option key={u.id} value={u.id}>{u.nombre}</option>)}</Select></div>
              <div><label className="text-xs font-semibold block mb-1">Ruta</label><Select value={form.rutaId} onChange={(e:any) => setForm({...form, rutaId: e.target.value})} disabled={!form.ufId}><option value="">Seleccione...</option>{rutas.map(r => <option key={r.id} value={r.id}>{r.nombre}</option>)}</Select></div>
            </div><div className="space-y-4">
              <div><label className="text-xs font-semibold block mb-1">Indicador</label><Select value={form.indicadorId} onChange={(e:any) => setForm({...form, indicadorId: e.target.value})}><option value="">Seleccione...</option>{state.indicadores.map(i => <option key={i.id} value={i.id}>{i.identificador} - {i.nombre}</option>)}</Select></div>
              <div><label className="text-xs font-semibold block mb-1">Fecha</label><Input type="date" value={form.fecha} onChange={(e:any) => setForm({...form, fecha: e.target.value})} /></div>
            </div></div>
          <div className="mt-8 flex justify-end"><Button onClick={() => {setStep(2); setCarrilesSeleccionados([]);}} disabled={!form.proyectoId || !form.ufId || !form.rutaId || !form.indicadorId || !form.fecha}>Siguiente <ChevronRight className="w-4 h-4 ml-1" /></Button></div>
        </Card>
      )}

      {step === 2 && (
        <Card className="p-6 animate-in fade-in">
          <h2 className="text-lg font-bold mb-2 border-b pb-2">2. Carriles</h2>
          <div className="grid grid-cols-2 gap-4 mb-6">{rutaActiva?.carrilesTags.map(tag => {

'border-[#00C2D4] bg-[#00C2D4]/5' : 'border-slate-200'}`}><div className={`w-6 h-6 rounded-full border-2 flex items-center justify-center shrink-0 ${isSelected ? 'border-[#00C2D4] bg-[#00C2D4] text-[#0A1628]' : 'border-slate-300 text-transparent'}`}><Check className="w-4 h-4" /></div><span className="font-bold text-lg">{tag}</span></div>);
            })}</div>
          <div className="mt-8 flex justify-between"><Button variant="outline" onClick={() => setStep(1)}><ChevronLeft className="w-4 h-4 mr-1" /> Anterior</Button><Button onClick={() => setStep(3)} disabled={carrilesSeleccionados.length === 0}>Siguiente <ChevronRight className="w-4 h-4 ml-1" /></Button></div>
        </Card>
      )}

      {step === 3 && (
        <Card className="p-6 animate-in fade-in">
          <h2 className="text-lg font-bold mb-6 border-b pb-2">3. Carga de Datos</h2>
          
          <div className="flex items-center justify-between mb-4">

  const [brushIndexes, setBrushIndexes] = useState<{startIndex: number, endIndex: number} | null>(null);
  const [fullscreenChart, setFullscreenChart] = useState<'line' | 'bar' | null>(null);

  const indicador = state.indicadores.find(i => i.id === medicionBase?.indicadorId);
  const condPuntual = indicador?.condiciones.find(c => c.tipo === 'PUNTUAL')?.valor || indicador?.condiciones.find(c => c.tipo === 'GENERAL')?.valor || 3.5;
  const condMedio   = indicador?.condiciones.find(c => c.tipo === 'MEDIO')?.valor || indicador?.condiciones.find(c => c.tipo === 'GENERAL')?.valor || 3.0;

  const medicionesPares = useMemo(() => {
    if (!medicionBase) return [];
    return state.mediciones.filter(m =>
      m.rutaId === medicionBase.rutaId &&
      m.indicadorId === medicionBase.indicadorId &&
      m.carrilesSeleccionados.some(t => medicionBase.carrilesSeleccionados.includes(t))
    ).sort((a, b) => a.fecha.localeCompare(b.fecha)); 
  }, [state.mediciones, medicionBase]);

  const medicionesConLabel = useMemo(() => {
    const dateGroups: Record<string, number> = {};
    medicionesPares.forEach(m => { dateGroups[m.fecha] = (dateGroups[m.fecha] || 0) + 1; });
    const counters: Record<string, number> = {};
    return medicionesPares.map(m => {
       if (dateGroups[m.fecha] > 1) {
          counters[m.fecha] = (counters[m.fecha] || 0) + 1;
          return { ...m, label: `${m.fecha} (Tramo ${counters[m.fecha]})` };
       }
       return { ...m, label: m.fecha };
    });
  }, [medicionesPares]);

  const [selectedMedIds, setSelectedMedIds] = useState<string[]>([]);
  
  const yearColorMap = useMemo(() => {
    const uniqueYears = Array.from(new Set(medicionesConLabel.map(m => m.fecha.substring(0, 4)))).sort();
    const map: Record<string, {izq: string, der: string}> = {};
    uniqueYears.forEach((year, idx) => { map[year] = COLOR_PAIRS[idx % COLOR_PAIRS.length]; });
    return map;
  }, [medicionesConLabel]);

  useEffect(() => {
    if (medicionBase && selectedMedIds.length === 0) setSelectedMedIds([medicionBase.id]);
  }, [medicionBase]);

  useEffect(() => { setBrushIndexes(null); }, [viewMode, selectedMedIds]);

  const toggleMedicion = (id: string) => setSelectedMedIds(prev => prev.includes(id) ? prev.filter(x => x !== id) : [...prev, id]);

  const combinedData = useMemo(() => {
    if (selectedMedIds.length === 0) return null;
    const allPuntuales: IRI_RegistroPuntual[] = [];
    const selectedMeds = medicionesConLabel.filter(m => selectedMedIds.includes(m.id));
    
    selectedMeds.forEach(m => allPuntuales.push(...m.datos.registros_puntual));
    if (allPuntuales.length === 0) return null;
    allPuntuales.sort((a,b) => a.from_m - b.from_m);

    const { stats, km } = agruparPorKm(allPuntuales, false, condPuntual, condMedio);
    if (!stats) return null;
    return { stats, km };
  }, [medicionesConLabel, selectedMedIds, condPuntual, condMedio]);

  const combinedDataCrudo = useMemo(() => {
    if (selectedMedIds.length === 0) return null;
    const allBaseRecords: IRI_RegistroBase[] = [];
    let missingData = false;
    selectedMedIds.forEach(id => {
      if (baseDataMap[id]) allBaseRecords.push(...baseDataMap[id]);
      else missingData = true; 
    });
    if (missingData && allBaseRecords.length === 0) return null;
    
    allBaseRecords.sort((a,b) => a.from_m - b.from_m);
    const { stats, km } = agruparPorKm(allBaseRecords, true, condPuntual, condMedio);
    if (!stats) return null;
    return { stats, km };
  }, [selectedMedIds, baseDataMap, condPuntual, condMedio]);

  const activeData = viewMode === '100m' ? combinedData : combinedDataCrudo;
  const safeActiveData = activeData || {
    stats: { valor_max_puntual: 0, valor_medio_global: 0, porcentaje_cumplimiento: 0, n_incumplimientos_puntual: 0, cumple_global: true, n_km_incumplen: 0 },
    km: [] as IRI_ResumenKm[]
  };

  useEffect(() => {
    if (viewMode === '20m') {
      const fetchMissing = async () => {
        setIsLoadingBase(true);
        const newMap = { ...baseDataMap };
        let changed = false;
        for (const id of selectedMedIds) {
          if (!newMap[id]) {
             const data = await getBaseRecords(id);
             newMap[id] = data;
             changed = true;
          }
        }
        if (changed) setBaseDataMap(newMap);
        setIsLoadingBase(false);
      };
      fetchMissing();
    }
  }, [viewMode, selectedMedIds]);

  const chartData100m = useMemo(() => {
    const abscisasSet = new Set<number>();
    const selectedMeds = medicionesConLabel.filter(m => selectedMedIds.includes(m.id));
    selectedMeds.forEach(m => m.datos.registros_puntual.forEach(r => abscisasSet.add(r.from_m)));
    
    return Array.from(abscisasSet).sort((a,b)=>a-b).map(abs => {
      const pt: any = { abscisa: abs };
      selectedMeds.forEach(m => {
        const r = m.datos.registros_puntual.find(x => x.from_m === abs);
        if (r) pt[m.id] = parseFloat(r.valor.toFixed(3));
      });
      return pt;
    });
  }, [medicionesConLabel, selectedMedIds]);

  const chartData20m = useMemo(() => {
    const abscisasSet = new Set<number>();
    const selectedMeds = medicionesConLabel.filter(m => selectedMedIds.includes(m.id));
    selectedMeds.forEach(m => {
      const bData = baseDataMap[m.id] || [];
      bData.forEach(r => abscisasSet.add(r.from_m));
    });
    
    return Array.from(abscisasSet).sort((a,b)=>a-b).map(abs => {
      const pt: any = { abscisa: abs };
      selectedMeds.forEach(m => {
        const bData = baseDataMap[m.id] || [];
        const r = bData.find(x => x.from_m === abs);
        if (r) {
          pt[`${m.id}_izq`] = r.iri_izq;
          pt[`${m.id}_der`] = r.iri_der;
        }
      });
      return pt;
    });
  }, [medicionesConLabel, selectedMedIds, baseDataMap]);

  const handleBarClick = useCallback((data: any) => {
    if (!data || !data.activePayload || data.activePayload.length === 0) return;
    const { from_m, to_m } = data.activePayload[0].payload;
    const currentChartData = viewMode === '100m' ? chartData100m : chartData20m;
    
    const startIdx = currentChartData.findIndex(d => d.abscisa >= from_m);
    let endIdx = currentChartData.findIndex(d => d.abscisa >= to_m);
    
    if (endIdx === -1) endIdx = currentChartData.length - 1;
    else if (endIdx > 0 && currentChartData[endIdx].abscisa > to_m) endIdx -= 1;
    
    if (startIdx !== -1) {
      setBrushIndexes({ startIndex: startIdx, endIndex: endIdx });
    }
  }, [viewMode, chartData100m, chartData20m]);

  // Lógica para escalas enteras amigables
  const getNiceMax = (val: number) => {
    const ceil = Math.ceil(val);
    if (ceil <= 4) return 4;
    if (ceil <= 6) return 6;
    if (ceil <= 8) return 8;
    if (ceil <= 10) return 10;
    return Math.ceil(ceil / 5) * 5;
  };

  const maxDomain100m = getNiceMax(Math.max(condPuntual * 1.5, safeActiveData.stats.valor_max_puntual * 1.1));

  const maxDomain20m = useMemo(() => {
     let max = 0;
     selectedMedIds.forEach(id => {
        (baseDataMap[id] || []).forEach(r => {
           if (r.iri_izq > max) max = r.iri_izq;
           if (r.iri_der > max) max = r.iri_der;
        });
     });
     return getNiceMax(Math.max(condPuntual * 1.5, max * 1.1));
  }, [baseDataMap, selectedMedIds, condPuntual]);

  const formatAbscisa = (val: number) => {
    const km   = Math.floor(val / 1000);
    const rest = String(val % 1000).padStart(3, '0');
    return `K${km}+${rest}`;
  };

  const CustomDot100m = (props: any) => {
    const { cx, cy, value, stroke } = props;
    if (value == null || cx == null || cy == null) return null;
    if (value > condPuntual) return <circle cx={cx} cy={cy} r={4} fill="#ef4444" stroke="none" />;
    return <circle cx={cx} cy={cy} r={2} fill={stroke} stroke="none" />;
  };

  const CustomDot20m = (props: any) => {
    const { cx, cy, value } = props;
    if (value == null || cx == null || cy == null) return null; 
    if (value > condPuntual) return <circle cx={cx} cy={cy} r={4} fill="#ef4444" stroke="none" />;
    return null;
  };

  if (!medicionBase || !indicador) return null;
  const proy  = state.proyectos.find(p => p.id === medicionBase.proyectoId);
  const uf    = state.unidadesFuncionales.find(u => u.id === medicionBase.unidadFuncionalId);
  const ruta  = state.rutas.find(r => r.id === medicionBase.rutaId);

  return (
    <div className="space-y-6 pb-20 animate-in fade-in">
      <div className="flex items-center gap-2 text-sm text-slate-500">
        <Button variant="ghost" size="sm" onClick={() => changeView('mediciones_list')}><ChevronLeft className="w-4 h-4 mr-1" /> Mediciones</Button>
        <ChevronRight className="w-3 h-3" /><span>{proy?.nombre}</span><ChevronRight className="w-3 h-3" /><span>{uf?.nombre}</span><ChevronRight className="w-3 h-3" /><span className="font-semibold text-[#1B3A6B]">{ruta?.nombre}</span>
      </div>

      <div className="flex justify-between items-end flex-wrap gap-4">
        <div>
          <h1 className="text-3xl font-bold mb-2 font-heading">
            {indicador.nombre}
            <span className="ml-3 text-lg text-slate-400 font-normal">[{medicionBase.carrilesSeleccionados.join(', ')}]</span>
          </h1>
          <div className="flex gap-2 flex-wrap">
            <Badge variant="blue">{indicador.identificador}</Badge>
            {activeData ? (
              <Badge variant={safeActiveData.stats.cumple_global ? 'green' : 'red'}>
                {safeActiveData.stats.cumple_global ? '✓ CUMPLE GLOBAL (Selección)' : '✗ INCUMPLE GLOBAL (Selección)'}
              </Badge>
            ) : (<Badge variant="default">Sin tramos seleccionados</Badge>)}
          </div>
        </div>
      </div>

      <div className="grid grid-cols-2 lg:grid-cols-4 gap-4">
        {[
          { label: `Valor máx. ${viewMode === '100m' ? 'puntual' : 'crudo'}`, value: safeActiveData.stats.valor_max_puntual.toFixed(2), unit: indicador.unidad_medida, ok: safeActiveData.stats.valor_max_puntual <= condPuntual },
          { label: `Valor medio global (${viewMode})`, value: safeActiveData.stats.valor_medio_global.toFixed(2), unit: indicador.unidad_medida, ok: safeActiveData.stats.valor_medio_global <= condMedio },
          { label: '% Cumplimiento', value: safeActiveData.stats.porcentaje_cumplimiento.toFixed(1), unit: '%', ok: safeActiveData.stats.porcentaje_cumplimiento >= 100 },
          { label: `Excesos (${viewMode === '100m' ? '100m' : 'Base'})`, value: String(safeActiveData.stats.n_incumplimientos_puntual), unit: 'puntos', ok: safeActiveData.stats.n_incumplimientos_puntual === 0 },
        ].map((card, i) => (
          <Card key={i} className={`p-4 border-l-4 ${card.ok ? 'border-l-emerald-400' : 'border-l-red-400 opacity-90'}`}>
            <p className="text-xs text-slate-500 mb-1">{card.label}</p>
            <p className={`text-2xl font-bold ${card.ok ? 'text-emerald-700' : 'text-red-600'}`}>{activeData ? card.value : '-'}</p>
            <p className="text-xs text-slate-400">{card.unit}</p>
          </Card>
        ))}
      </div>

      <Card className={fullscreenChart === 'line' ? "fixed inset-0 z-[60] bg-white p-6 shadow-2xl flex flex-col" : "p-6"}>
        <div className="flex justify-between items-start mb-6 flex-wrap gap-4">
          <div>
            <h3 className="font-bold text-lg font-heading"><TrendingUp className="inline w-5 h-5 text-[#00C2D4] mr-2" />Perfil de Medición</h3>
            <p className="text-xs text-slate-500 mt-0.5">Límite Analizado: {condPuntual} {indicador.unidad_medida}</p>
          </div>

          <div className="flex items-center gap-4">
            {brushIndexes && (
               <Button variant="outline" size="sm" onClick={() => setBrushIndexes(null)} className="h-7 text-xs border-[#00C2D4] text-[#00C2D4]"><Maximize className="w-3 h-3 mr-1" /> Restaurar Vista</Button>
            )}
            <div className="flex bg-slate-100 p-1 rounded-md">
              <button className={`px-3 py-1 text-xs rounded transition-all ${viewMode === '100m' ? 'bg-white shadow text-[#00C2D4] font-semibold' : 'text-slate-500'}`} onClick={() => setViewMode('100m')}>Puntual (100m)</button>
              <button className={`px-3 py-1 text-xs rounded transition-all ${viewMode === '20m' ? 'bg-white shadow text-[#00C2D4] font-semibold' : 'text-slate-500'}`} onClick={() => setViewMode('20m')}>Crudo ({medicionBase.datos.intervalo_base_m}m) {isLoadingBase && '...'}</button>
            </div>
            <Button variant="ghost" size="sm" onClick={() => setFullscreenChart(fullscreenChart === 'line' ? null : 'line')} className="h-7 text-xs text-slate-500 hover:text-slate-800 shrink-0">
               {fullscreenChart === 'line' ? <><Minimize className="w-3 h-3 mr-1" /> Salir</> : <><Maximize className="w-3 h-3 mr-1" /> Pantalla Completa</>}
            </Button>
          </div>
        </div>
        
        <div className="bg-blue-50/50 p-3 rounded-md border border-blue-100 text-sm text-slate-600 mb-6 flex items-start gap-2">
          <Info className="w-5 h-5 text-blue-500 shrink-0" />
          <p><strong>Nota técnica:</strong> Algunos indicadores normativos evalúan de manera distinta el <strong>Valor Puntual (promedios en tramos más cortos)</strong> y el <strong>Valor Medio (cada 1km)</strong>. Los "datos crudos" representan la medición base pura registrada por el equipo; si observas puntos rojos en la vista cruda, indica que esa lectura bruta excedió el límite máximo permitido por la norma.</p>
        </div>

        {/* GRÁFICA 100m multi-tramo */}
        {viewMode === '100m' && (
          <div className={fullscreenChart === 'line' ? "flex-1 w-full min-h-0" : "h-[450px] w-full"}>
            {selectedMedIds.length === 0 ? (
               <div className="flex items-center justify-center h-full text-slate-400">Selecciona al menos un tramo para visualizar la gráfica.</div>
            ) : (
            <ResponsiveContainer width="100%" height="100%">
              <ComposedChart data={chartData100m} margin={{ top: 10, right: 30, left: 10, bottom: 20 }}>
                <CartesianGrid strokeDasharray="3 3" vertical={false} stroke="#f1f5f9" />
                <XAxis dataKey="abscisa" tickFormatter={formatAbscisa} tick={{ fontSize: 11, fill: '#64748b' }} minTickGap={60} />
                <YAxis domain={[0, maxDomain100m]} tick={{ fontSize: 11, fill: '#64748b' }} />
                <RechartsTooltip
                  formatter={(value: any, name: string) => {
                     const m = medicionesConLabel.find(x => x.id === name);
                     return [`${Number(value).toFixed(3)} ${indicador.unidad_medida}`, m ? m.label : name];
                  }}
                  labelFormatter={(label) => `Abscisa: ${formatAbscisa(label as number)}`}
                  contentStyle={{ fontSize: 12, borderRadius: 8 }}
                />
                
                <Brush 
                  dataKey="abscisa" height={30} stroke="#00C2D4" tickFormatter={formatAbscisa} 
                  startIndex={brushIndexes ? brushIndexes.startIndex : 0} 
                  endIndex={brushIndexes ? brushIndexes.endIndex : chartData100m.length - 1} 
                  onChange={(e: any) => setBrushIndexes({startIndex: e.startIndex, endIndex: e.endIndex})}
                />

                <ReferenceArea y1={condPuntual} y2={maxDomain100m} fill="#fee2e2" fillOpacity={0.4} />
                <ReferenceLine y={condPuntual} stroke="#ef4444" strokeDasharray="6 3" strokeWidth={1.5} label={{ value: `Límite ${condPuntual}`, position: 'insideTopRight', fontSize: 11, fill: '#ef4444' }} />
                
                {medicionesConLabel.filter(m => selectedMedIds.includes(m.id)).map((mObj) => {
                  const year = mObj.fecha.substring(0, 4);
                  const colorBase = yearColorMap[year]?.der || '#000';
                  return <Line key={mObj.id} type="monotone" dataKey={mObj.id} name={mObj.id} stroke={colorBase} strokeWidth={2.5} dot={<CustomDot100m />} activeDot={{ r: 6, strokeWidth: 2 }} connectNulls={false} />;
                })}
                <Legend verticalAlign="top" height={36} formatter={(value) => { const m = medicionesConLabel.find(x => x.id === value); return m ? m.label : value; }} />
              </ComposedChart>
            </ResponsiveContainer>
            )}
          </div>
        )}

        {/* GRÁFICA BASE (20m o la leída) multi-tramo */}
        {viewMode === '20m' && (
          <div className={fullscreenChart === 'line' ? "flex-1 w-full min-h-0" : "h-[450px] w-full"}>
            {selectedMedIds.length === 0 ? (
               <div className="flex items-center justify-center h-full text-slate-400">Selecciona al menos un tramo para visualizar la gráfica.</div>
            ) : (
              <ResponsiveContainer width="100%" height="100%">
                <ComposedChart data={chartData20m} margin={{ top: 10, right: 30, left: 10, bottom: 20 }}>
                  <CartesianGrid strokeDasharray="3 3" vertical={false} stroke="#f1f5f9" />
                  <XAxis dataKey="abscisa" tickFormatter={formatAbscisa} tick={{ fontSize: 11 }} minTickGap={60} />
                  <YAxis domain={[0, maxDomain20m]} tick={{ fontSize: 11 }} />
                  <RechartsTooltip
                    formatter={(v: any, name: string) => [`${Number(v).toFixed(3)} ${indicador.unidad_medida}`, name]}
                    labelFormatter={(label) => `Abscisa: ${formatAbscisa(label as number)}`}
                    contentStyle={{ fontSize: 12, borderRadius: 8 }}
                  />
                  
                  <Brush 
                    dataKey="abscisa" height={30} stroke="#00C2D4" tickFormatter={formatAbscisa} 
                    startIndex={brushIndexes ? brushIndexes.startIndex : 0} 
                    endIndex={brushIndexes ? brushIndexes.endIndex : chartData20m.length - 1} 
                    onChange={(e: any) => setBrushIndexes({startIndex: e.startIndex, endIndex: e.endIndex})}
                  />

                  <ReferenceArea y1={condPuntual} y2={maxDomain20m} fill="#fee2e2" fillOpacity={0.4} />
                  <ReferenceLine y={condPuntual} stroke="#ef4444" strokeDasharray="6 3" strokeWidth={1.5} label={{ value: `Límite ${condPuntual}`, position: 'insideTopRight', fontSize: 11, fill: '#ef4444' }} />
                  
                  {medicionesConLabel.filter(m => selectedMedIds.includes(m.id)).map((mObj) => {
                     const year = mObj.fecha.substring(0, 4);
                     const pair = yearColorMap[year] || COLOR_PAIRS[0];
                     return (
                        <React.Fragment key={mObj.id}>
                          <Line type="monotone" dataKey={`${mObj.id}_izq`} name={`Medición Izquierda - ${mObj.label}`} stroke={pair.izq} strokeWidth={2} dot={<CustomDot20m />} activeDot={{ r: 4 }} connectNulls={false} />
                          <Line type="monotone" dataKey={`${mObj.id}_der`} name={`Medición Derecha - ${mObj.label}`}  stroke={pair.der} strokeWidth={2} strokeDasharray="4 4" dot={<CustomDot20m />} activeDot={{ r: 4 }} connectNulls={false} />
                        </React.Fragment>
                     );
                  })}
                  <Legend verticalAlign="top" height={36} />
                </ComposedChart>
              </ResponsiveContainer>
            )}
          </div>
        )}
      </Card>

      <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
        <Card className={fullscreenChart === 'bar' ? "fixed inset-0 z-[60] bg-white p-6 shadow-2xl flex flex-col" : "p-4 flex flex-col"}>
          <div className="flex justify-between items-start mb-4">
             <div>
                <h3 className="font-bold text-sm mb-1"><BarChart2 className="inline w-4 h-4 mr-1 text-[#00C2D4]" />Valor Medio por Km (Evaluación Normativa)</h3>
                <p className="text-xs text-slate-500 max-w-2xl">Promedio aritmético de las medidas en segmentos de 1,000m (basado en la abscisa inicial). Límite: ≤ {condMedio} {indicador.unidad_medida}. <br/><span className="text-blue-500 font-semibold">Haz clic en una barra para hacer zoom en la gráfica superior.</span></p>
             </div>
             <Button variant="ghost" size="sm" onClick={() => setFullscreenChart(fullscreenChart === 'bar' ? null : 'bar')} className="h-7 text-xs text-slate-500 hover:text-slate-800 shrink-0">
               {fullscreenChart === 'bar' ? <><Minimize className="w-3 h-3 mr-1" /> Salir</> : <><Maximize className="w-3 h-3 mr-1" /> Expandir</>}
            </Button>
          </div>
          <div className={fullscreenChart === 'bar' ? "flex-1 min-h-0 w-full" : "flex-1 min-h-[250px] w-full"}>
            {safeActiveData.km.length > 0 ? (
            <ResponsiveContainer width="100%" height="100%">
              <BarChart data={safeActiveData.km} margin={{ top: 10, right: 10, left: -20, bottom: 20 }} onClick={handleBarClick}>
                <XAxis dataKey="km_label" tickFormatter={(val) => val.split(' ')[0]} tick={{ fontSize: 10 }} />
                <YAxis tick={{ fontSize: 10 }} />
                <RechartsTooltip formatter={(v: any) => [`${Number(v).toFixed(2)} ${indicador.unidad_medida}`, 'Valor medio']} cursor={{ fill: 'transparent' }} contentStyle={{ fontSize: 12, borderRadius: 8 }} />
                <ReferenceLine y={condMedio} stroke="#f59e0b" strokeDasharray="4 2" />
                <Brush dataKey="km_label" height={30} stroke="#00C2D4" tickFormatter={(val) => val.split(' ')[0]} />
                <Bar dataKey="valor_medio" radius={[3, 3, 0, 0]} className="cursor-pointer hover:opacity-80 transition-opacity">
                  {safeActiveData.km.map((entry, index) => <Cell key={index} fill={entry.cumple_medio ? '#10b981' : '#ef4444'} /> )}
                </Bar>
              </BarChart>
            </ResponsiveContainer>
            ) : (
              <div className="flex items-center justify-center h-full text-slate-400 text-xs">Sin datos</div>
            )}
          </div>
        </Card>
        
        <Card className="lg:col-span-2 flex flex-col overflow-hidden">
          <div className="p-3 border-b flex justify-between items-center bg-slate-50">
            <h3 className="font-bold text-sm">Detalle de Incumplimientos y Promedios (Tramos Combinados)</h3>
            <span className="text-xs text-slate-500">{safeActiveData.stats.n_km_incumplen} km incumplen / {safeActiveData.km.length} total</span>
          </div>
          <div className="flex-1 overflow-auto max-h-[300px]">
            <table className="w-full text-xs text-left">
              <thead className="bg-white sticky top-0 font-semibold shadow-sm">
                <tr>
                  <th className="p-3">Km (Desde abscisa real)</th>
                  <th className="p-3">Valor Medio</th>
                  <th className="p-3 text-center">Incumplimientos ({viewMode === '100m' ? '100m' : 'Crudos'})</th>
                  <th className="p-3">Estado Km</th>
                </tr>
              </thead>
              <tbody className="divide-y">
                {safeActiveData.km.map((km, i) => (
                  <tr key={i} className={!km.cumple_km ? 'bg-red-50/40' : ''}>
                    <td className="p-3 font-medium">{km.km_label}</td>
                    <td className="p-3 font-bold">{km.valor_medio.toFixed(2)}</td>
                    <td className="p-3 text-center">{km.n_incumplimientos > 0 ? <span className="inline-flex items-center gap-1 text-red-600 font-bold"><AlertCircle className="w-3 h-3" /> {km.n_incumplimientos}</span> : <span className="text-slate-400">0</span>}</td>
                    <td className="p-3"><Badge variant={km.cumple_km ? 'green' : 'red'}>{km.cumple_km ? 'CUMPLE' : 'INCUMPLE'}</Badge></td>
                  </tr>
                ))}
                {safeActiveData.km.length === 0 && <tr><td colSpan={4} className="p-4 text-center text-slate-400">Sin datos</td></tr>}
              </tbody>
            </table>
          </div>
        </Card>
      </div>
    </div>
  );
};

// --- INICIALIZACIÓN PRINCIPAL ---
export default function App() {
  const [state, dispatch] = useReducer(appReducer, initialState, (initial) => {
    try {
      const item = window.localStorage.getItem('aqua_infra_state');
      return item ? JSON.parse(item) : initial;
    } catch { return initial; }
  });

  const [viewProps, setViewProps] = useState<any>({});
  
  const changeView = useCallback((view: string, props?: any) => {
    setViewProps(props || {});
    dispatch({ type: 'SET_MODULE', payload: view });
  }, []);

  useEffect(() => {
    try {
      window.localStorage.setItem('aqua_infra_state', JSON.stringify(state));
    } catch (error) {
      console.warn("Storage cap reached or error.", error);
    }
  }, [state]);

  return (
    <AppContext.Provider value={{ state, dispatch, changeView, viewProps }}>
      <AppLayout />
    </AppContext.Provider>
  );
}
